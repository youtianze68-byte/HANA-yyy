  <button type="button" onclick="evolve()">進化させる</button>
  
  <div id="message">ボタンを押して進化を開始！</div>

  <script>
    let level = 0;
    const messages = [
      "通常のはなさん。",
      "クール度が20%上昇。",
      "オーラが見え始めた。",
      "周囲が静まり返る。",
      "後輩がひれ伏す。",
      "伝説クラス到達。",
      "次のステージへ進学完了。",
      "卒業おめでとう。最強。"
    ];

    function createPetals(num) {
      for(let i=0; i<num; i++) {
        const p = document.createElement('div');
        p.className = 'petal';
        p.style.width = p.style.height = Math.random()*10+5 + 'px';
        p.style.left = Math.random()*100 + 'vw';
        p.style.animationDuration = (Math.random()*2+2) + 's';
        document.body.appendChild(p);
        setTimeout(() => p.remove(), 4000);
      }
    }

    function evolve() {
      if (level >= messages.length) {
        document.getElementById("message").innerText = "これ以上進化できません。完成体です。";
        return;
      }

      level++;
      const msgDiv = document.getElementById("message");
      const hana = document.getElementById("hana");
      const aura = document.getElementById("aura");
      const cap = document.getElementById("cap");
      const bg = document.getElementById("bg");

      // メッセージ更新
      msgDiv.innerText = messages[level-1];

      // 大きさを変更
      hana.style.transform = `scale(${1 + (level * 0.15)})`;

      // 演出の追加
      if (level >= 3) aura.style.opacity = "1";
      
      if (level >= 5) {
        bg.classList.add("vibrate");
        setTimeout(() => bg.classList.remove("vibrate"), 200);
      }

      createPetals(level * 4);

      // 最終形態（レベル8）の特別演出
      if (level === 8) {
        cap.style.display = "block";
        bg.style.background = "#1a000d";
        msgDiv.style.color = "#ff69b4";
        createPetals(150);
      }
    }
  </script>
